CREATE OR REPLACE FUNCTION count_tcs_high_salary
RETURN NUMBER
IS
    v_count NUMBER := 0;
BEGIN 
    SELECT COUNT(*)
    INTO v_count
    FROM EMPLOYEE
    WHERE COMPANY='TCS' 
        AND SALARY > 50000;

    RETURN v_count;
END;
/
